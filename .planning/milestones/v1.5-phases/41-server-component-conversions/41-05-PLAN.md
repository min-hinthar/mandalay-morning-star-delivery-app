**Strategy:**
HomePageClient's main purpose is SectionNavDots (scroll spy) which requires client-side scroll tracking.

Options:

1. **Keep HomePageClient minimal** - only for SectionNavDots, pass all sections as children
2. **Move static sections to page.tsx** - render directly in server component
3. **Create composition pattern** - server page.tsx composes, HomePageClient wraps for nav

Recommended approach:

```typescript
// page.tsx (Server Component)
import { Suspense } from 'react'
import { getFeaturedSections } from '@/lib/queries/sections'
import { Hero } from '@/components/ui/homepage/Hero'
import { TestimonialsCarousel } from '@/components/ui/homepage/TestimonialsCarousel'
import { CTABanner } from '@/components/ui/homepage/CTABanner'
import { FooterCTA } from '@/components/ui/homepage/FooterCTA'
import { HomepageMenuSection } from '@/components/ui/homepage/HomepageMenuSection'
import { HomePageWrapper } from '@/components/ui/homepage/HomePageWrapper'

// Skeletons for Suspense
function MenuSkeleton() { /* existing */ }
function HowItWorksSkeleton() { /* existing */ }

// Lazy load heavy component
const HowItWorksSection = lazy(() => import('@/components/ui/homepage/HowItWorksSection'))

export default async function HomePage() {
  const featuredSections = await getFeaturedSections()

  return (
    <main className="min-h-screen bg-background">
      {/* HomePageWrapper provides SectionNavDots */}
      <HomePageWrapper>
        <Hero ctaHref="/menu" />

        <Suspense fallback={<HowItWorksSkeleton />}>
          <HowItWorksSection id="how-it-works" />
        </Suspense>

        <Suspense fallback={<MenuSkeleton />}>
          <HomepageMenuSection featuredSections={featuredSections} />
        </Suspense>

        <TestimonialsCarousel id="testimonials" />
        <CTABanner id="cta" />
        <FooterCTA />
      </HomePageWrapper>
    </main>
  )
}
```

```typescript
// HomePageWrapper.tsx (Client Component - minimal)
'use client'
import { SectionNavDots } from '@/components/ui/scroll'

const sections = [
  { id: 'hero', label: 'Home' },
  { id: 'how-it-works', label: 'How It Works' },
  { id: 'menu', label: 'Menu' },
  { id: 'testimonials', label: 'Reviews' },
  { id: 'cta', label: 'Order' },
]

export function HomePageWrapper({ children }: { children: React.ReactNode }) {
  return (
    <>
      <SectionNavDots sections={sections} />
      {children}
    </>
  )
}
```

**Key changes:**

- Move section composition to server page.tsx
- HomePageWrapper is tiny client component just for nav dots
- Sections render as server components where possible
- Suspense boundaries for async/lazy content

If this causes hydration issues due to animation components expecting client context, revert to simpler approach.
