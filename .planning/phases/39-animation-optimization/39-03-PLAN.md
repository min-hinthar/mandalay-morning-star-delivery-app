3. Update click handler to use FlyToCart callbacks:
   ```typescript
   const handleClick = useCallback((e: React.MouseEvent<HTMLButtonElement>) => {
     e.stopPropagation();

     // Get source element for fly animation
     const sourceEl = e.currentTarget;

     // Trigger fly animation with callbacks
     const success = fly({
       sourceElement: sourceEl,
       imageUrl: item?.image_url || undefined,
       onAnimationStart: () => {
         // Show checkmark while flying
         setShowCheckmark(true);
         // Clear any existing timeout
         if (checkmarkTimeoutRef.current) {
           clearTimeout(checkmarkTimeoutRef.current);
         }
       },
       onAnimationComplete: () => {
         // Hide checkmark after brief delay (~500ms per CONTEXT.md)
         checkmarkTimeoutRef.current = setTimeout(() => {
           setShowCheckmark(false);
         }, 500);
       },
     });

     // Add to cart (optimistic - happens immediately)
     onAdd();
   }, [fly, item?.image_url, onAdd]);
   ```

4. Update button content to show checkmark when `showCheckmark` is true:
   ```tsx

   )}
   ```

5. Add cleanup in useEffect:
   ```typescript
   useEffect(() => {
     return () => {
       if (checkmarkTimeoutRef.current) {
         clearTimeout(checkmarkTimeoutRef.current);
       }
   ```
