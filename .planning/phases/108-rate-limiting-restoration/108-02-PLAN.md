Replace the entire `checkRedis` function with:

```typescript
export async function checkRedis(): Promise<ServiceStatus> {
  const configured = Boolean(
    process.env.UPSTASH_REDIS_REST_URL && process.env.UPSTASH_REDIS_REST_TOKEN
  );

  if (!configured) {
    return { status: "down", configured: false };
  }

  const start = Date.now();
  try {
    const { getRedisClient } = await import("@/lib/rate-limit");
    const client = getRedisClient();
    if (!client) {
      // D-38: Redis unavailable but app works via in-memory fallback
      return { status: "degraded", configured: true, connected: false };
    }
    // D-40: 3-second timeout via Promise.race (AbortSignal not supported by @upstash/redis HTTP client)
    const pingResult = await Promise.race([
      client.ping(),
      new Promise<never>((_, reject) =>
        setTimeout(() => reject(new Error("PING timeout (3s)")), 3000)
      ),
    ]);
    // D-39: Include latency in response
    return {
      status: "healthy",
      configured: true,
      connected: true,
      latency_ms: Date.now() - start,
    };
  } catch (err) {
    // D-38: Redis down = degraded (not "down"), app still works via in-memory fallback
    return {
      status: "degraded",
      configured: true,
      connected: false,
      latency_ms: Date.now() - start,
      error: err instanceof Error ? err.message : "unknown",
    };
  }
}
```

Key design points:
- Uses `Promise.race` for 3s timeout per Pitfall 6 (AbortSignal not supported by @upstash/redis)
- Reports "degraded" not "down" when Redis fails per D-38 (app works via in-memory fallback)
- Dynamic import of `@/lib/rate-limit` to avoid circular dependency
- `pingResult` variable captures the return but is unused (ping is for connectivity, not data)
- Add `// eslint-disable-next-line @typescript-eslint/no-unused-vars` above the pingResult line, or just use `await Promise.race(...)` without assignment
