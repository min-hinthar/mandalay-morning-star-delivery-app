**PreferenceCounterCard.tsx:**

Per CONTEXT.md: "Simple aggregate counts on admin side (e.g., '12 customers with nut allergy')."
Per discretion recommendation: "Place as a MetricCard on admin dashboard alongside existing metrics."

Create a server component (no `"use client"` needed — data fetched server-side):

```typescript
interface PreferenceCounterCardProps {
  className?: string;
}

export async function PreferenceCounterCard({ className }: PreferenceCounterCardProps) {
  // Fetch aggregate counts from customer_settings
  const supabase = await createClient();

  // Query all customer_settings rows
  const { data: allSettings } = await supabase
    .from("customer_settings")
    .select("dietary_restrictions, notification_prefs");

  // Count dietary restriction occurrences
  // dietary_restrictions is JSONB array like ["vegetarian", "gluten-free"]
  const dietaryCounts: Record<string, number> = {};
  const notifOptOuts = { order_updates: 0, marketing: 0, reminders: 0 };
  let totalCustomers = 0;

  for (const row of allSettings ?? []) {
    totalCustomers++;
    const restrictions = row.dietary_restrictions as string[];
    for (const r of restrictions) {
      dietaryCounts[r] = (dietaryCounts[r] || 0) + 1;
    }
    const prefs = row.notification_prefs as Record<string, boolean>;
    if (!prefs.order_updates) notifOptOuts.order_updates++;
    if (!prefs.marketing) notifOptOuts.marketing++;
    if (!prefs.reminders) notifOptOuts.reminders++;
  }

  // Render as a Card
}
```
