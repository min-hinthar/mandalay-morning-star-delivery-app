    Per CONTEXT.md - Detailed stop card:
    ```tsx
    <motion.div className="bg-surface-secondary rounded-card-sm border p-4">
      {/* Header: Stop number, Status badge, ETA */}
      <div className="flex justify-between items-start">
        <div className="flex items-center gap-2">
          <span className="font-bold text-lg">{stopIndex + 1}</span>
          <StatusBadge status={stop.status} />
        </div>
        <span className="text-sm text-text-muted">ETA: {formatTime(stop.eta)}</span>
      </div>

      {/* Customer Info */}
      <div className="mt-3">
        <p className="font-medium">{stop.order.customer.fullName}</p>
        <p className="text-sm text-text-secondary">{stop.order.address.line1}</p>
        <p className="text-sm text-text-muted">{stop.order.address.city}</p>
      </div>

      {/* Contact Buttons - click to call/SMS */}
      <div className="flex gap-2 mt-3">
        <a href={`tel:${stop.order.customer.phone}`}>
          <Button size="sm" variant="outline">Call</Button>
        </a>
        <a href={`sms:${stop.order.customer.phone}`}>
          <Button size="sm" variant="outline">SMS</Button>
        </a>
      </div>

      {/* Order Summary */}
      <div className="mt-3 p-2 bg-surface-tertiary rounded-input">
        <p className="text-sm">{stop.order.itemCount} items • ${(stop.order.totalCents/100).toFixed(2)}</p>
        {stop.order.specialInstructions && (
          <p className="text-xs text-text-muted mt-1">Note: {stop.order.specialInstructions}</p>
        )}
      </div>

      {/* Delivery Window */}
      {stop.order.deliveryWindowStart && (
        <p className="text-xs text-text-muted mt-2">
          Window: {formatTime(start)} - {formatTime(end)}
        </p>
      )}

      {/* Proof Photo (if delivered) */}
      {stop.deliveryPhotoUrl && (
        <img src={stop.deliveryPhotoUrl} className="mt-3 rounded-input w-20 h-20 object-cover cursor-pointer" />
      )}

      {/* Exception Badge */}
      {stop.exception && !stop.exception.resolved && (
        <div className="mt-3 p-2 bg-status-error/10 border border-status-error/20 rounded-input">
          <Badge variant="destructive">{stop.exception.type}</Badge>
          <p className="text-xs mt-1">{stop.exception.description}</p>
        </div>
      )}

      {/* Status Change Dropdown (admin override) */}
      <DropdownMenu>
        <DropdownMenuTrigger asChild>
          <Button size="sm" variant="ghost" className="mt-3">Change Status</Button>
        </DropdownMenuTrigger>
        <DropdownMenuContent>
          {['pending', 'enroute', 'arrived', 'delivered', 'skipped'].map(s => (
            <DropdownMenuItem key={s} onClick={() => onStatusChange(stop.id, s)}>
              {s}
            </DropdownMenuItem>
          ))}
        </DropdownMenuContent>
      </DropdownMenu>
    </motion.div>
    ```
