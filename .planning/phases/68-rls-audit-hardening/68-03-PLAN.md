```javascript
let failures = 0;
function assert(condition, message) {
  if (!condition) {
    console.error(`FAIL: ${message}`);
    failures++;
  } else {
    console.log(`PASS: ${message}`);
  }
}
```

2. Query `driver_badges` as Driver A. Store badges.
3. Sign out Driver A, sign in as Driver B.
4. Query `routes` as Driver B. Assert Driver B cannot see Driver A's routes: `assert(!routesB.some(r => r.driver_id === driverAId), "Driver B cannot see Driver A routes")`.
