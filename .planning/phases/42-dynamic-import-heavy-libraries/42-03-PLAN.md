    Create two lazy map wrappers. Both use `LoadingWithTimeout` wrapping `MapSkeleton` with `timeoutMs={15000}` (15 seconds -- longer than charts' 10s to account for mobile networks per CONTEXT.md):

    **LazyRouteMap:**
    ```tsx
    export const LazyRouteMap = dynamic(
      () => importWithRetry(
        () => import("@/components/ui/admin/routes/RouteMap").then(mod => mod.RouteMap),
        "RouteMap"
      ),
      {
        loading: () => (
          <LoadingWithTimeout
            skeleton={<MapSkeleton height={400} />}
            timeoutMs={15000}
            timeoutMessage="Map taking longer than expected"
          />
        ),
        ssr: false,
      }
    );
    ```
    Height 400 matches the `h-[400px]` container in RouteDetailClient.

    **LazyDeliveryMap:**
    ```tsx
    export const LazyDeliveryMap = dynamic(
      () => importWithRetry(
        () => import("@/components/ui/orders/tracking/DeliveryMap").then(mod => mod.DeliveryMap),
        "DeliveryMap"
      ),
      {
        loading: () => (
          <LoadingWithTimeout
            skeleton={<MapSkeleton height={300} />}
            timeoutMs={15000}
            timeoutMessage="Map taking longer than expected"
          />
        ),
        ssr: false,
      }
    );
    ```
