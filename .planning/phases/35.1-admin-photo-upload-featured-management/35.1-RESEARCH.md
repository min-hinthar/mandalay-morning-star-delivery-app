### Pattern 1: Supabase Storage Upload with RLS

**What:** Admin-only file upload to public bucket
**When to use:** Photo uploads from admin dashboard
**Example:**

```typescript
// Source: Existing pattern from src/app/api/driver/routes/[routeId]/stops/[stopId]/photo/route.ts
const MAX_FILE_SIZE = 10 * 1024 * 1024; // 10MB per CONTEXT.md
const VALID_TYPES = ["image/jpeg", "image/png"]; // JPEG + PNG only per CONTEXT.md

async function uploadMenuPhoto(file: File, menuItemId: string) {
  // Validate file
  if (file.size > MAX_FILE_SIZE) throw new Error("File too large");
  if (!VALID_TYPES.includes(file.type)) throw new Error("Invalid file type");

  // Generate path: menu-photos/{menu_item_id}.{ext}
  const ext = file.type === "image/png" ? "png" : "jpg";
  const filename = `${menuItemId}.${ext}`;

  // Upload (upsert to replace existing)
  const { error } = await supabase.storage.from("menu-photos").upload(filename, file, {
    contentType: file.type,
    upsert: true,
  });

  if (error) throw error;

  // Get public URL
  const { data } = supabase.storage.from("menu-photos").getPublicUrl(filename);

  return data.publicUrl;
}
```

### Pattern 3: Admin RLS for Storage

**What:** Storage bucket policies using is_admin() function
**When to use:** Menu photos bucket security
**Example:**

```sql
-- Source: Pattern from supabase/migrations/004_storage.sql
INSERT INTO storage.buckets (id, name, public, file_size_limit, allowed_mime_types)
VALUES (
  'menu-photos',
  'menu-photos',
  true,  -- Public read, RLS controls write
  10485760, -- 10MB
  ARRAY['image/jpeg', 'image/png']
);

-- Only admin can upload
CREATE POLICY "menu_photos_insert" ON storage.objects
FOR INSERT TO authenticated
WITH CHECK (
  bucket_id = 'menu-photos' AND public.is_admin()
);

-- Public read (bucket is public)
CREATE POLICY "menu_photos_select" ON storage.objects
FOR SELECT USING (bucket_id = 'menu-photos');

-- Only admin can delete
CREATE POLICY "menu_photos_delete" ON storage.objects
FOR DELETE TO authenticated
USING (bucket_id = 'menu-photos' AND public.is_admin());
```

### Anti-Patterns to Avoid

- **Don't store images in DB:** Use Supabase Storage, not base64 in database
- **Don't skip RLS:** Always enforce admin-only write access
- **Don't use JSONB for section items:** Lose foreign key constraints, harder to query
- **Don't auto-refresh on every change:** Use explicit publish action per CONTEXT.md

## Don't Hand-Roll

| Problem              | Don't Build         | Use Instead                     | Why                                                              |
| -------------------- | ------------------- | ------------------------------- | ---------------------------------------------------------------- |
| Image resize/crop    | Canvas manipulation | Supabase Storage transforms     | Handled server-side, cached, no client-side processing           |
| Drag-and-drop        | Custom mouse events | @dnd-kit                        | Keyboard accessibility, touch support, collision detection       |
| File upload progress | XHR progress events | fetch + progress from file.size | Supabase doesn't expose upload progress; estimate from file size |

| Alt text generation  | Custom logic        | Menu item name                  | Per CONTEXT.md: "Auto-generate alt text from menu item name"     |

## Common Pitfalls

### Pitfall 1: Storage RLS Recursion

**What goes wrong:** Using `auth.uid()` in storage policies causes infinite recursion
**Why it happens:** Storage policies that join with profiles table trigger RLS on profiles
**How to avoid:** Use `public.is_admin()` SECURITY DEFINER function (already exists)
**Warning signs:** Error: "infinite recursion detected in policy"

### Pitfall 2: Missing Cascade Delete Triggers

**What goes wrong:** Menu item deleted but photo remains in storage
**Why it happens:** No automatic cleanup of storage objects
**How to avoid:** Create database trigger on menu_items DELETE to call storage delete
**Warning signs:** Orphaned photos accumulating in storage bucket

### Pitfall 3: Large File Upload Timeout

**What goes wrong:** 10MB uploads fail on slow connections
**Why it happens:** Default fetch timeout too short
**How to avoid:** Use chunked upload or increase timeout; show clear progress UI
**Warning signs:** "Failed to fetch" errors on large files only

### Pitfall 4: Hydration Mismatch in Featured Sections

**What goes wrong:** Server/client render different section orders
**Why it happens:** Auto-populated "Most Popular" uses dynamic data
**How to avoid:** Fetch order data server-side; cache with revalidation
**Warning signs:** Console warnings about hydration mismatch

### Pitfall 5: Drag-Drop Without Optimistic Updates

**What goes wrong:** Laggy UI when reordering sections
**Why it happens:** Waiting for DB round-trip before updating UI
**How to avoid:** Optimistic update state immediately, rollback on error
**Warning signs:** Visual "jump back" after drag completion

### Photo Upload Hook

```typescript
// src/lib/hooks/usePhotoUpload.ts
import { useState, useCallback } from "react";
import { createClient } from "@/lib/supabase/client";
import { toast } from "@/lib/hooks/useToast";

const MAX_FILE_SIZE = 10 * 1024 * 1024;
const VALID_TYPES = ["image/jpeg", "image/png"];

interface UploadState {
  uploading: boolean;
  progress: number;
  error: string | null;
}

export function usePhotoUpload(bucketName: string = "menu-photos") {
  const [state, setState] = useState<UploadState>({
    uploading: false,
    progress: 0,
    error: null,
  });

  const upload = useCallback(
    async (file: File, path: string): Promise<string | null> => {
      // Validate
      if (file.size > MAX_FILE_SIZE) {
        setState({ uploading: false, progress: 0, error: "File too large (max 10MB)" });
        return null;
      }
      if (!VALID_TYPES.includes(file.type)) {
        setState({ uploading: false, progress: 0, error: "Only JPEG and PNG allowed" });
        return null;
      }

      setState({ uploading: true, progress: 0, error: null });
      const supabase = createClient();

      try {
        // Simulated progress (Supabase doesn't expose real progress)
        const progressInterval = setInterval(() => {
          setState((prev) => ({ ...prev, progress: Math.min(prev.progress + 10, 90) }));
        }, 200);

        const { error } = await supabase.storage
          .from(bucketName)
          .upload(path, file, { contentType: file.type, upsert: true });

        clearInterval(progressInterval);

        if (error) throw error;

        const { data } = supabase.storage.from(bucketName).getPublicUrl(path);
        setState({ uploading: false, progress: 100, error: null });
        return data.publicUrl;
      } catch (err) {
        const message = err instanceof Error ? err.message : "Upload failed";
        setState({ uploading: false, progress: 0, error: message });
        toast({ title: "Upload failed", description: message, variant: "destructive" });
        return null;
      }
    },
    [bucketName]
  );

  return { ...state, upload };
}
```

### Sortable Section List with @dnd-kit

```typescript
// Example sortable list pattern
import {
  DndContext,
  closestCenter,
  KeyboardSensor,
  PointerSensor,
  useSensor,
  useSensors,
  DragEndEvent,
} from "@dnd-kit/core";
import {
  arrayMove,
  SortableContext,
  sortableKeyboardCoordinates,
  verticalListSortingStrategy,
  useSortable,
} from "@dnd-kit/sortable";
import { CSS } from "@dnd-kit/utilities";

function SortableSection({ section }: { section: FeaturedSection }) {
  const { attributes, listeners, setNodeRef, transform, transition } = useSortable({
    id: section.id,
  });

  const style = {
    transform: CSS.Transform.toString(transform),
    transition,
  };

  return (
    <div ref={setNodeRef} style={style} {...attributes} {...listeners}>
      {/* Section card content */}
    </div>
  );
}

function SectionList({ sections, onReorder }: Props) {
  const sensors = useSensors(
    useSensor(PointerSensor),
    useSensor(KeyboardSensor, { coordinateGetter: sortableKeyboardCoordinates })
  );

  function handleDragEnd(event: DragEndEvent) {
    const { active, over } = event;
    if (over && active.id !== over.id) {
      const oldIndex = sections.findIndex(s => s.id === active.id);
      const newIndex = sections.findIndex(s => s.id === over.id);
      onReorder(arrayMove(sections, oldIndex, newIndex));
    }
  }

  return (
    <DndContext sensors={sensors} collisionDetection={closestCenter} onDragEnd={handleDragEnd}>
      <SortableContext items={sections.map(s => s.id)} strategy={verticalListSortingStrategy}>
        {sections.map(section => (
          <SortableSection key={section.id} section={section} />
        ))}
      </SortableContext>
    </DndContext>
  );
}
```
