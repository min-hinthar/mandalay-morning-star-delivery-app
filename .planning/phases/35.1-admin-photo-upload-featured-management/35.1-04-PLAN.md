import { createClient } from '@/lib/supabase/server';
import type { FeaturedSectionWithItems } from '@/types/featured-sections';
import type { MenuItem } from '@/types/menu';

export async function getFeaturedSections(): Promise<FeaturedSectionWithItems[]> {
  const supabase = await createClient();

  // Get visible, non-deleted sections ordered by sort_order
  const { data: sections, error: sectionsError } = await supabase
    .from('featured_sections')
    .select('*')
    .eq('is_visible', true)
    .is('deleted_at', null)
    .order('sort_order', { ascending: true });

  if (sectionsError) throw sectionsError;
  if (!sections) return [];

  // Get items for each section
  const sectionsWithItems = await Promise.all(
    sections.map(async (section) => {
      const { data: junctionRows } = await supabase
        .from('featured_section_items')
        .select('item_id, sort_order')
        .eq('section_id', section.id)
        .order('sort_order', { ascending: true })
        .limit(section.item_count);

      if (!junctionRows || junctionRows.length === 0) {
        return { ...mapSectionRow(section), items: [] };
      }

      const itemIds = junctionRows.map((r) => r.item_id);

      const { data: items } = await supabase
        .from('menu_items')
        .select(`
          *,
          menu_categories!inner(id, name, slug)
        `)
        .in('id', itemIds)
        .eq('is_active', true);

      // Map to MenuItem type and preserve sort order
      const mappedItems = (items || [])
        .map((item) => mapMenuItemRow(item))
        .sort((a, b) => {
          const aIndex = itemIds.indexOf(a.id);
          const bIndex = itemIds.indexOf(b.id);
          return aIndex - bIndex;
        });

      return { ...mapSectionRow(section), items: mappedItems };
    })
  );

  // Filter out empty sections
  return sectionsWithItems.filter((s) => s.items.length > 0);
}

// Helper to map database row to FeaturedSection type
function mapSectionRow(row: any): FeaturedSection {
  return {
    id: row.id,
    slug: row.slug,
    name: row.name,
    subtitle: row.subtitle,
    icon: row.icon,
    accentColor: row.accent_color,
    sortOrder: row.sort_order,
    itemCount: row.item_count,
    isVisible: row.is_visible,
    isPredefined: row.is_predefined,
    deletedAt: row.deleted_at,
    createdAt: row.created_at,
    updatedAt: row.updated_at,
    updatedBy: row.updated_by,
  };
}

1. `src/components/ui/icons/DynamicIcon.tsx` - Dynamic lucide icon loader:

Create a utility component that renders lucide icons by name string:

```typescript
import { LucideIcon, icons, Sparkles } from 'lucide-react';
import { ComponentProps } from 'react';

interface DynamicIconProps extends ComponentProps<LucideIcon> {
  name: string;
  fallback?: LucideIcon;
}

export function DynamicIcon({ name, fallback = Sparkles, ...props }: DynamicIconProps) {
  // Convert kebab-case to PascalCase for lucide icon lookup
  const pascalName = name
    .split('-')
    .map(part => part.charAt(0).toUpperCase() + part.slice(1))
    .join('');

  const IconComponent = icons[pascalName as keyof typeof icons] || fallback;

  if (!IconComponent) {
    return null;
  }

  return <IconComponent {...props} />;
}
```

This enables rendering icons from database-stored icon names like "utensils", "flame", "sparkles".

2. `FeaturedSections.tsx` - Container for all featured sections:

```typescript
interface FeaturedSectionsProps {
  sections: FeaturedSectionWithItems[];
  onItemSelect: (item: MenuItem) => void;
  favorites: Set<string>;
  onFavoriteToggle: (item: MenuItem, isFavorite: boolean) => void;
}

export function FeaturedSections({
  sections,
  onItemSelect,
  favorites,
  onFavoriteToggle,
}: FeaturedSectionsProps) {
  if (sections.length === 0) return null;

  return (
    <div className="space-y-12 md:space-y-16">
      {sections.map((section, index) => (
        <motion.div
          key={section.id}
          initial={{ opacity: 0, y: 20 }}
          whileInView={{ opacity: 1, y: 0 }}
          viewport={{ once: true, margin: "-80px" }}
          transition={spring.gentle}
        >
          {/* Section Header */}
          <div className="flex items-center gap-3 mb-6 px-4 md:px-6">
            {section.icon && (
              <div
                className="flex items-center gap-2 px-3 py-1.5 rounded-full"
                style={{ backgroundColor: `${section.accentColor}20` || 'var(--secondary-10)' }}
              >
                <DynamicIcon name={section.icon} className="w-4 h-4" style={{ color: section.accentColor }} />
                <span className="text-sm font-body font-semibold" style={{ color: section.accentColor }}>
                  {section.name}
                </span>
              </div>
            )}
            {!section.icon && (
              <h3 className="font-display text-xl md:text-2xl font-bold text-primary">
                {section.name}
              </h3>
            )}
            {section.subtitle && (
              <p className="text-sm font-body text-text-muted">{section.subtitle}</p>
            )}
            <Link
              href={`/menu?section=${section.slug}`}
              className="ml-auto text-sm font-body text-primary hover:underline"
            >
              View All
            </Link>
          </div>

          {/* First section: Grid layout, others: Carousel */}
          {index === 0 ? (
            <SectionGrid
              items={section.items}
              onItemSelect={onItemSelect}
              favorites={favorites}
              onFavoriteToggle={onFavoriteToggle}
            />
          ) : (
            <SectionCarousel
              items={section.items}
              onItemSelect={onItemSelect}
              favorites={favorites}
              onFavoriteToggle={onFavoriteToggle}
              accentColor={section.accentColor}
            />
          )}
        </motion.div>
      ))}
    </div>
  );
}
```
