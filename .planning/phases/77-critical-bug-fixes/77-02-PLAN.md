2. Fix `getCutoffForSaturday()` to use `CUTOFF_DAY`:
```typescript
export function getCutoffForSaturday(saturday: Date): Date {
  const { year, month, day } = getZonedParts(saturday);
  const utcDate = new Date(Date.UTC(year, month - 1, day));

  // CUTOFF_DAY is day-of-week (5 = Friday). Saturday is 6.
  // Offset = 6 - CUTOFF_DAY = days before Saturday
  const daysBeforeSaturday = 6 - CUTOFF_DAY;
  utcDate.setUTCDate(utcDate.getUTCDate() - daysBeforeSaturday);

  return zonedTimeToUtc({
    year: utcDate.getUTCFullYear(),
    month: utcDate.getUTCMonth() + 1,
    day: utcDate.getUTCDate(),
    hour: CUTOFF_HOUR,
    minute: 0,
    second: 0,
  });
}
```

Note: The current hardcoded `-1` happens to be correct for CUTOFF_DAY=5 (Friday is 1 day before Saturday). But using the constant makes it configurable and correct for any cutoff day.
