```typescript
import { test, expect } from "@playwright/test";

/**
 * Cart Flow E2E Tests (Phase 47)
 *
 * Closes deferred Phase 43 requirements:
 * - REQ-43.4: Cart on customer routes
 * - REQ-43.8: Deep linking to cart routes
 * - REQ-43.9: No cart flow regressions
 */

test.describe("Cart Flow - Happy Path (REQ-43.4)", () => {
  test("add item to cart", async ({ page }) => {
    await page.goto("/");
    await page.locator('[data-testid="menu-item"]').first().click();
    await page.getByRole("button", { name: /add to cart/i }).click();

    // Verify cart indicator updated
    const cartButton = page.locator('[data-testid="cart-button"]');
    await expect(cartButton).toContainText("1");
  });

  test("modify quantity in cart", async ({ page }) => {
    // Add item first
    await page.goto("/");
    await page.locator('[data-testid="menu-item"]').first().click();
    await page.getByRole("button", { name: /add to cart/i }).click();
    await page.keyboard.press("Escape");

    // Open cart drawer
    await page.locator('[data-testid="cart-button"]').click();

    // Increase quantity
    const increaseBtn = page.getByRole("button", { name: /\+|increase/i });
    if (await increaseBtn.isVisible()) {
      await increaseBtn.click();
      // Cart should now show 2
      await expect(page.locator('[data-testid="cart-button"]')).toContainText("2");
    }
  });

  test("remove item from cart", async ({ page }) => {
    // Add item first
    await page.goto("/");
    await page.locator('[data-testid="menu-item"]').first().click();
    await page.getByRole("button", { name: /add to cart/i }).click();
    await page.keyboard.press("Escape");

    // Open cart drawer
    await page.locator('[data-testid="cart-button"]').click();

    // Remove item
    const removeBtn = page.getByRole("button", { name: /remove|delete/i }).first();
    if (await removeBtn.isVisible()) {
      await removeBtn.click();
      // Cart should be empty or show 0
      await expect(page.getByText(/empty|no items/i)).toBeVisible();
    }
  });

  test("proceed to checkout", async ({ page }) => {
    // Add item first
    await page.goto("/");
    await page.locator('[data-testid="menu-item"]').first().click();
    await page.getByRole("button", { name: /add to cart/i }).click();
    await page.keyboard.press("Escape");

    // Navigate to checkout via cart
    await page.locator('[data-testid="cart-button"]').click();
    await page.getByRole("button", { name: /checkout/i }).click();

    // Should be on checkout or login page
    await expect(page).toHaveURL(/(checkout|login)/);
  });
});

test.describe("Cart Flow - Edge Cases (REQ-43.4)", () => {
  test("empty cart shows empty state", async ({ page }) => {
    await page.goto("/cart");
    await expect(page.getByText(/empty|no items/i)).toBeVisible();
  });

  test("cart persists across navigation", async ({ page }) => {
    // Add item
    await page.goto("/");
    await page.locator('[data-testid="menu-item"]').first().click();
    await page.getByRole("button", { name: /add to cart/i }).click();
    await page.keyboard.press("Escape");

    // Navigate to menu
    await page.goto("/menu");

    // Cart should still have item
    const cartButton = page.locator('[data-testid="cart-button"]');
    await expect(cartButton).toContainText("1");
  });
});

test.describe("Deep Link Verification (REQ-43.8)", () => {
  test("/cart deep link loads correctly", async ({ page }) => {
    await page.goto("/cart");
    // Should show cart page (empty or with items)
    await expect(page).toHaveURL(/cart/);
  });

  test("/checkout with empty cart redirects to /menu", async ({ page }) => {
    await page.goto("/checkout");
    // Should redirect to menu or login
    await expect(page).toHaveURL(/(menu|login)/);
  });

  test("/menu/[id] allows add to cart", async ({ page }) => {
    // Navigate to a menu item detail page
    await page.goto("/");
    await page.locator('[data-testid="menu-item"]').first().click();

    // Should be able to add to cart
    const addButton = page.getByRole("button", { name: /add to cart/i });
    await expect(addButton).toBeVisible();
  });
});

test.describe("Cart Regression Check (REQ-43.9)", () => {
  test("CartIndicator opens drawer on cart routes", async ({ page }) => {
    // Add item first
    await page.goto("/");
    await page.locator('[data-testid="menu-item"]').first().click();
    await page.getByRole("button", { name: /add to cart/i }).click();
    await page.keyboard.press("Escape");

    // Click cart indicator
    await page.locator('[data-testid="cart-button"]').click();

    // Should open drawer (dialog/sheet)
    const drawer = page.getByRole("dialog");
    await expect(drawer).toBeVisible();
  });

  test("FlyToCart animation triggers on add", async ({ page }) => {
    await page.goto("/");
    await page.locator('[data-testid="menu-item"]').first().click();
    await page.getByRole("button", { name: /add to cart/i }).click();

    // Cart indicator should update (animation completed)
    const cartButton = page.locator('[data-testid="cart-button"]');
    await expect(cartButton).toContainText("1");
  });
});
```
