    **Reorder flow with explicit cart store wiring:**
    ```typescript
    import { useCartStore } from '@/stores/cart-store';

    const handleReorder = async (orderId: string) => {
      setReordering(orderId);
      try {
        const response = await fetch(`/api/account/orders/${orderId}/reorder`, { method: 'POST' });
        const data = await response.json();

        if (!response.ok) throw new Error(data.error);

        // Explicit cart store mutation
        const cartStore = useCartStore.getState();
        cartStore.clearCart(); // Optional: clear existing cart
        data.cartItems.forEach((item: CartItem) => {
          cartStore.addItem({
            menuItemId: item.menuItemId,
            name: item.name,
            quantity: item.quantity,
            priceCents: item.priceCents,
            modifiers: item.modifiers || [],
          });
        });

        // Show warnings if any items unavailable/price changed
        if (data.warnings?.length > 0) {
          toast.warning(`${data.warnings.length} item(s) had issues - check your cart`);
        } else {
          toast.success('Items added to cart');
        }

        router.push('/cart');
      } catch (error) {
        toast.error('Failed to reorder');
      } finally {
        setReordering(null);
      }
    };
    ```

    **Cancel flow:**
    ```typescript
    const handleCancel = async (orderId: string) => {
      // Show confirmation dialog with reason input
      const reason = await showCancelDialog(); // Returns reason or null if cancelled
      if (!reason) return;

      try {
        const response = await fetch(`/api/account/orders/${orderId}/cancel`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ reason }),
        });

        if (!response.ok) {
          const data = await response.json();
          throw new Error(data.error);
        }

        toast.success('Order cancelled');
        // Refresh orders list
        mutate();
      } catch (error) {
        toast.error(error.message || 'Failed to cancel order');
      }
    };
    ```
